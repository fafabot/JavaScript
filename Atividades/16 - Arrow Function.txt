let dobro = (numero) => {
    return numero * 2
}

console.log("\n")

let somar = (a, b) => {

console.log("\n")

let maiorNumero = (a, b) => {
    return a > b ? a : b
}

let num7 = Number(prompt("Digite o segundo número:"))

console.log("6 - Verificar Par ou Ímpar;\n")
let parOuImpar = (numero) => {
    return numero % 2 === 0 ? "Par" : "Ímpar"
}
let num9 = Number(prompt("Digite um número:"))
console.log(parOuImpar(num9))


console.log("\n")

console.log("7 - Verificar Número Positivo ou Negativo;\n")
let verificarNumero = (numero) => {
    if (numero > 0) {
        return "Positivo"
    } else if (numero < 0) {
        return "Negativo"
    } else {
        return "Zero"
    }
}
let num8 = Number(prompt("Digite um número:"))
console.log(verificarNumero(num8))


console.log("\n")

console.log("8 - Calcular Média;\n")
let media = (a, b, c) => {
    return (a + b + c) / 3
}
let n1 = Number(prompt("Digite a primeira nota:"))
let n2 = Number(prompt("Digite a segunda nota:"))
let n3 = Number(prompt("Digite a terceira nota:"))
console.log("Média:", media(n1, n2, n3))


console.log("\n")

console.log("10 - Verificar Aprovação;\n")
let verificarAprovacao = (nota1, nota2) => {
    let mediaFinal = (nota1 + nota2) / 2

    if (mediaFinal >= 7) {
        return "Aprovado"
    } else if (mediaFinal >= 5) {
        return "Recuperação"
    } else {
        return "Reprovado"
    }
}
let nota1 = Number(prompt("Digite a primeira nota:"))
let nota2 = Number(prompt("Digite a segunda nota:"))
console.log(verificarAprovacao(nota1, nota2))

console.log("\n")
